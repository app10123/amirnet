        function renderSCList() {
            let listView = document.getElementById('sc-list-view');
            if (!listView) return;
            let total = sentenceQuestions.length;
            let correctCount = Object.values(sentenceProgress).filter(p => p.correct).length;
            let wrongCount = Object.values(sentenceProgress).filter(p => p.answered && !p.correct).length;
            let unseenCount = total - correctCount - wrongCount;
            let pct = Math.round((correctCount / total) * 100);

            let gridHtml = sentenceQuestions.map((q, i) => {
                let prog = sentenceProgress[i];
                let cls = 'sc-cell';
                let label = i + 1;
                if (prog && prog.answered) {
                    cls += prog.correct ? ' sc-correct' : ' sc-wrong';
                    label = prog.correct
                        ? `<svg class="sc-check-svg" viewBox="0 0 16 16"><polyline class="sc-check-path" points="2,9 6,13 14,4"/></svg>`
                        : `<svg class="sc-check-svg" viewBox="0 0 16 16"><line class="sc-x-path" x1="3" y1="3" x2="13" y2="13"/><line class="sc-x-path" x1="13" y1="3" x2="3" y2="13"/></svg>`;
                }
                return `<button class="${cls}" onclick="showSCQuestion(${i})">${label}</button>`;
            }).join('');

            listView.innerHTML = `
                <div class="sc-progress-box">
                    <div class="progress-bar-bg" style="margin-bottom:10px;"><div class="progress-bar-fill" style="width:${pct}%"></div></div>
                    <div class="stats-text">
                        <span class="stat-known"><svg viewBox="0 0 14 14" style="width:13px;height:13px;display:inline-block;vertical-align:middle;margin-left:2px;"><polyline points="2,8 5,11 12,3" fill="none" stroke="currentColor" stroke-width="2.5" stroke-linecap="round" stroke-linejoin="round"/></svg> נכון: ${correctCount}</span>

        function _ttsSetIcon(btn, speaking, withLabel) {
            if (!btn) return;
            if (withLabel) {
                btn.innerHTML = speaking ? (_TTS_WAVES + ' ' + withLabel) : (_TTS_ICON + ' ' + withLabel);
            } else {
                btn.innerHTML = speaking ? _TTS_WAVES : _TTS_ICON;
            }
        }
        let _ttsBtn = null;
        let _ttsBtnLabel = null;
        function ttsStop() {
            if (window.speechSynthesis.speaking) window.speechSynthesis.cancel();
            if (_ttsBtn) { _ttsSetIcon(_ttsBtn, false, _ttsBtnLabel); _ttsBtn.classList.remove('speaking'); _ttsBtn = null; _ttsBtnLabel = null; }
        }
        function ttsSpeak(texts, btnEl, rowLabel) {
            if (window.speechSynthesis.speaking) {
                let prevBtn = _ttsBtn;
                ttsStop();
                if (prevBtn === btnEl) return;
            }
            _ttsBtn = btnEl;
            _ttsBtnLabel = rowLabel || null;
            if (btnEl) { _ttsSetIcon(btnEl, true, rowLabel); btnEl.classList.add('speaking'); }
            const run = (idx) => {
                if (idx >= texts.length) { _ttsSetIcon(btnEl, false, rowLabel); if (btnEl) btnEl.classList.remove('speaking'); _ttsBtn = null; _ttsBtnLabel = null; return; }
                let utt = new SpeechSynthesisUtterance(texts[idx]);
                utt.lang = 'en-US';
                utt.rate = getTTSRate();
                utt.onend = () => run(idx + 1);
                utt.onerror = () => { _ttsSetIcon(btnEl, false, rowLabel); if (btnEl) btnEl.classList.remove('speaking'); _ttsBtn = null; _ttsBtnLabel = null; };
                window.speechSynthesis.speak(utt);
            };
            run(0);
        }

        function ttsPassage(btnEl) {
            let el = document.getElementById('rc-content');
            let raw = el.innerText.trim();
            let sentences = raw.split(/(?<=[.?!])\s+/).map(s => s.trim()).filter(s => s.length > 0);
            if (sentences.length === 0) sentences = [raw];
            ttsSpeak(sentences, btnEl, 'הקרא את הקטע');
        }

        function renderSentenceQuestion() {
            let container = document.getElementById('sc-container');
            let i = currentSentenceIndex;
            let q = sentenceQuestions[i];
            let total = sentenceQuestions.length;
            let prog = sentenceProgress[i];
            let isAnswered = prog && prog.answered;

            let optionsHtml = q.options.map((opt, optIndex) => {
                let cls = 'q-opt-btn';
                let disabled = '';
                if (isAnswered) {
                    disabled = 'disabled';
                    if (optIndex === q.correct) cls += ' correct';
                    else if (prog.selectedOpt === optIndex) cls += ' wrong';
                }
                let click = isAnswered ? '' : `onclick="answerSC(${i}, ${optIndex}, ${q.correct})"`;
                return `<button class="${cls}" ${disabled} ${click}>${opt}</button>`;
            }).join('');

            let expStyle = isAnswered ? 'display:block' : 'display:none';
            let nextLabel = i < total - 1 ? 'השאלה הבאה ›' : '← סיום — חזור לרשימה';

            container.innerHTML = `
                <div class="sc-nav-row">
                    <button class="sc-nav-btn" onclick="showSCQuestion(${i - 1})" ${i === 0 ? 'disabled' : ''}>
                        <svg viewBox="0 0 24 24" fill="none" stroke-width="2.5" stroke-linecap="round"><polyline points="15 18 9 12 15 6"></polyline></svg>
                    </button>
                    <span style="font-weight:800; color:#64748b; font-size:15px; direction:rtl;">שאלה ${i + 1} / ${total}</span>
                    <button class="sc-nav-btn" onclick="showSCQuestion(${i + 1})" ${i === total - 1 ? 'disabled' : ''}>
                        <svg viewBox="0 0 24 24" fill="none" stroke-width="2.5" stroke-linecap="round"><polyline points="9 18 15 12 9 6"></polyline></svg>
                    </button>
                </div>
                ${timerHTML('sc')}

                    <div style="display:flex;align-items:flex-start;gap:8px;margin-bottom:6px;">
                        <div class="sentence-text-q" style="margin-top:0;flex:1;">${q.q}</div>
                        <button class="tts-btn" onclick="ttsSC(this)" title="הקרא משפט" style="display:flex;align-items:center;justify-content:center;">${_TTS_ICON}</button>

                    <button class="tts-btn tts-row-btn" onclick="ttsSCOpts(this)" style="display:flex;align-items:center;gap:8px;">${_TTS_ICON} הקרא תשובות</button>

        function answerSC(qIndex, selectedOpt, correctOpt) {
            stopSCTimer();
            let isCorrect = selectedOpt === correctOpt;
            if (!sentenceProgress[qIndex]) {
                let ds = JSON.parse(localStorage.getItem('dailyStats')) || {};
                let today = new Date().toISOString().split('T')[0];
                ds[today] = (ds[today] || 0) + 1;
                localStorage.setItem('dailyStats', JSON.stringify(ds));
                renderChart();
            }
            sentenceProgress[qIndex] = { answered: true, correct: isCorrect, selectedOpt: selectedOpt };
            localStorage.setItem('sentenceProgress', JSON.stringify(sentenceProgress));

            let btns = document.getElementById(`sc-opts-${qIndex}`).querySelectorAll('.q-opt-btn');
            btns.forEach(b => b.disabled = true);
            if (isCorrect) {
                btns[selectedOpt].classList.add('correct');
                fireConfetti();
            } else {
                btns[selectedOpt].classList.add('wrong');
                btns[correctOpt].classList.add('correct');
            }
            document.getElementById(`sc-exp-${qIndex}`).style.display = 'block';
        }

        function nextSentence() {
            if (currentSentenceIndex < sentenceQuestions.length - 1) {
                showSCQuestion(currentSentenceIndex + 1);
            } else {
                showSCList();
            }
        }

        function resetSentenceProgress() {
            if (confirm('לאפס את כל ההתקדמות בהשלמת משפטים?')) {
                sentenceProgress = {};
                localStorage.removeItem('sentenceProgress');
                renderSCList();
            }
        }
